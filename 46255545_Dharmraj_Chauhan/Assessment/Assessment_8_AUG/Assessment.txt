1) REMOVE SIMILAR STRING FROM THE ARRAY.
---------------------------------------------------------------------------------------------------------------------------------------

#include <stdio.h>
#include<string.h>

int main()
{
    int n = 6, i, j, k;
    char str[6][6] = {"user1", "user2", "user3", "user4", "user1", "user3"};
    
    size_t total =  (sizeof(str)/sizeof(str[0]));
    for(i = 0; i < total; i++) {
        for(j = i+1; j < total; j++) {
            if(strcmp(str[i], str[j]) == 0) {
                for(k = 0; k < total; k++) {
                    str[j][k] = '\0';
                }
            }
        }
    }
    for(i = 0; i < total; i++) {
        for(j = 0; j < total; j++) {
            printf("%c", str[i][j]);
        }
        printf("\n");
    }
    return 0;
}
