/*Program to accept two digits and a character from the command line and
display the and store it in file*/
#include <stdio.h>
int main(int arg,char *argv[]) {
	int operator, x, y;

	if(arg<4){
		printf("invalid \n");
	}
	else{
		x=atoi(argv[1]);
		y=atoi(argv[2]);

		char *c=argv[3];
		FILE *fp;
		char *operator="operations.txt";
		fp=fopen(operator,"a");

		switch(*c)
		{
			case '+':
				fprintf(fp,"Sum of %d and %d is : %d \n ", x, y, x+y);
				break;
			case '-':
				fprintf(fp,"Difference of %d and %d is : %d \n", x, y, x-y);
				break;
			case 'x':
				fprintf(fp,"Multiplication of %d and %d is : %d \n", x, y, x*y);
				break;
			case '/':
				fprintf(fp,"Division of %d and %d is : %d \n", x, y, x/y);
				break;
			case '%':
				fprintf(fp,"Modulo of %d and %d is : %d \n", x, y, x%y);
				break;
			case '&':    
				fprintf(fp,"AND operation of %d and %d is : %d \n", x, y, x&y);
				break;
			case '|':
				fprintf(fp,"OR operation of %d and %d is : %d \n", x, y, x|y);
				break;
			case '^':
				fprintf(fp,"XOR operation of %d and %d is : %d \n", x, y, x^y);
				break;
			default :
				printf(fp," Enter Your Correct Choice.\n");
				break;
		}
		fclose(fp);
	}
	return 0;
}
