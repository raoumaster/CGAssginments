#include <stdio.h>
#include<stdlib.h>
#include<string.h>
#define EXIT_SUCCESS 0
#define EXIT_FAILURE 1
#define MAX 30

/*------swapping first element with element at given index--------*/


void swap_elements(char (*ptr)[MAX], int postion){
        char *first = *ptr;
        ptr = ptr + postion;
        char *second = *ptr;
        char *temp = NULL;

        printf("\nElement:\n");
        printf("First = %s, Second = %s\n", first, second);

        temp = first;
        first = second;
        second = temp;

        printf("\n Element After Swapping:\n");
        printf("First = %s, Second = %s\n", first, second);
}

//--------------------------------checking for vowel or non-vowel-------------------------------------------


int isVowel(char ch)
{
        if(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' ||
                        ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U')
                return EXIT_SUCCESS;
        else
                return EXIT_FAILURE;
}



//-------------remove all vowel characters and shift the remaining non-vowel characters-------------------------




char* remove_vowel(char (*ptr)[MAX], int index){
        ptr = ptr + index;
        char str[MAX] = "";
        strcpy(str, *ptr);

        char rmv_str[MAX];
        int j = 0;

        for (int i = 0; str[i] != '\0'; i++) {
                if (isVowel(str[i]) == 1) { 
                        rmv_str[j] = str[i];
                        j++;
                }
        }
        rmv_str[j] = '\0';
        strcpy(*ptr, rmv_str);

        return *ptr;
}





//----------------concating two strings-------------------------------





char *mystrcat(char *s1, char *s2){
        int i;
        for(i = 0; s1[i] != '\0'; i++);
        for(int j = 0; s2[j] != '\0'; j++, i++)
        {
                s1[i] = s2[j];
        }
        s1[i] = '\0';

        return s1;
}




//---------------------------display the contents of array--------------------------------------

        for(int i = 0; i < 4; i++)
