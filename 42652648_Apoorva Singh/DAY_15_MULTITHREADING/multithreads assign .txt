/* code to  make 2 threads 
   count number of words & char */
#include<stdio.h>
#include<stdlib.h>
#include<pthread.h>
#include<unistd.h>
#include<string.h>
#define MAX 100

void *words(void *str){
        int w=1;
        char *arr = (char *) str;
        for(int i=0; arr[i] != '\0'; i++){
                if(arr[i] != ' ' && arr[i+1] == ' '){
                        w++;
                }
        }
        printf("Words: %d\n",w);
}

void *chr(void *str){
        int c = 0;
        char *arr = (char *) str;
        for(int i=0; arr[i] != '\0'; i++){
                if(arr[i] != ' ')
                        c++;
        }
        printf("Characters: %d\n",c);
}

int main(){
        char str[MAX];
        printf("Enter String: ");
        fgets(str, MAX, stdin);

        pthread_t t1;
        pthread_t t2;

        pthread_create(&t1, NULL, &words, str);
        pthread_create(&t2, NULL, &chr, str);

        pthread_join(t1,NULL);
        pthread_join(t2,NULL);
}
