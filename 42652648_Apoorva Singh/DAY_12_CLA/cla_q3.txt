
// Function to extract Server name and Port number from command line argument.
 
#include <stdio.h>
#include <string.h>
#include <ctype.h>

int *disp(int argc, char *argv[]){
        for(int i=1; i<argc; i++){
                char *ptr = strtok(argv[i], ":");
                printf("Server Name Arg[%d]: %s\n",i,ptr);
                ptr = strtok(NULL, ":");

                printf("Port Number for Arg[%d]: %s\n\n",i,ptr);
        }
}

int main(int argc,char* argv[]){
        if (argc < 4)
                printf("Arguments less than 3\n");
        else
                disp(argc, argv);
        return 0;
}
