#include<stdio.h>
#include<pthread.h>
#include<unistd.h>
#include<stdlib.h>


#define MAX 100
char lin[MAX];
int count =0;
int wordcount =1;
void * charactercount(void *thr_p){

        char *arr=(char *)thr_p;f
        
        for(int i =0 ; arr[i] != '\0';i++){
                   if(arr[i] != ' '){
                        count++;
                   }
                }

        printf("%s",arr);
       pthread_exit(NULL);

}

void *wordcounts(void *thr_w){
        char *arr=(char *)thr_w;
        
        for(int i = 0;arr[i] !='\0';i++){
                if(arr[i] == ' '){
                        wordcount++;
                }
        }
        
        pthread_exit(NULL);
}

int main(){
        pthread_t tidOne; 
        pthread_t tidtwo;
        printf("enter the line: ");
        fgets(lin,MAX,stdin);

        pthread_create(&tidOne,NULL,&charactercount,lin);
        pthread_create(&tidtwo,NULL,&wordcounts,lin);
       
        pthread_join(tidOne,NULL);
        pthread_join(tidtwo,NULL);
         printf("\n the number of character is: %d",count-1);
         printf("\n the number of word is :%d ",wordcount);

}
