int main(int argc, char*argv[])
{
        FILE *myfile;

        pthread_t thread1;
        pthread_t thread2;
        pthread_t thread3;

        myfile = fopen(argv[1], "r");

        if(myfile == NULL);
        {

                printf("Not open file");
                exit (0);

        }

        else

        {



                pthread_create(&thread1, NULL, charCount, myfile);
                pthread_join(thread1, NULL);

                pthread_create(&thread2, NULL, wordCount, myfile);
                pthread_join(thread2, NULL);

                pthread_create(&thread3, NULL, lineCount, myfile);
                pthread_join(thread3, NULL);

        }

        fclose(myfile);
        return 0;
}
