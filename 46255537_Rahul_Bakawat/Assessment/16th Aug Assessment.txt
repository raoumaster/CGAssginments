/* Assessment 2 - Q4
 * Program to accept two digits and a character from the command line 
 * and display the result and store it in the file
 * such that everytime the code is executed the new operations is not overwritten
 */

#include <stdio.h>
int main(int arg,char *argv[]) {
	int op, a, b;

	if(arg<4){
		printf("invalid \n");
	}
	else{
		a=atoi(argv[1]);
		b=atoi(argv[2]);

		char *c=argv[3];

		FILE *fp;
		char *op="operations.txt";
		fp=fopen(op,"a");

		switch(*c)
		{
			case '+':
				fprintf(fp,"Sum of %d and %d is : %d \n ", a, b, a+b);
				break;
			case '-':
				fprintf(fp,"Difference of %d and %d is : %d \n", a, b, a-b);
				break;
			case 'x':
				fprintf(fp,"Multiplication of %d and %d is : %d \n", a, b, a*b);
				break;
			case '/':
				fprintf(fp,"Division of %d and %d is : %d \n", a, b, a/b);
				break;
			case '%':
				fprintf(fp,"Modulo of %d and %d is : %d \n", a, b, a%b);
				break;
			case '&':    
				fprintf(fp,"AND operation of %d and %d is : %d \n", a, b, a&b);
				break;
			case '|':
				fprintf(fp,"OR operation of %d and %d is : %d \n", a, b, a|b);
				break;
			case '^':
				fprintf(fp,"XOR operation of %d and %d is : %d \n", a, b, a^b);
				break;
			default :
				printf(fp," Enter Your Correct Choice.\n");
				break;
		}
		fclose(fp);

	}
	return 0;
}
