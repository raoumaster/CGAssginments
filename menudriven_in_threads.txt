#include<stdio.h>
#include<pthread.h>
#include<stdlib.h>
void *fib()
{
        int n,sum,i=2;
        int a=0,b=1;
        printf("Enter value of n:");
        scanf("%d",&n);
        printf("%d\t%d\t",a,b);
        while(i<n)
        {
                sum=a+b;
                printf("%d\t",sum);
                a=b;
                b=sum;
                i++;
        }
        printf("\n");
}
void *fact()
{
        int n,i;
        long int fact=1;
        printf("Enter the value of n:");
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        {
                fact*=i;
        }
        printf("factorial of %d is %ld\n",n,fact);
}
void *prime()
{
        int n,i,flag=0;
        printf("Enter the value of n:");
        scanf("%d",&n);
        for(i=2;i<=n;i++)
        {
               if(n%i==0)
                {
                        flag=1;
                        break;
                }
        }
        if(flag==0)
                printf("%d is prime\n",n);
        else
                printf("%d is not prime\n",n);
}
int main()
{
        pthread_t tid1;
        pthread_t tid2;
        pthread_t tid3;
        int ch;
        while(ch<=4)
        {
        printf("1.Fib\n");
        printf("2.fact\n");
        printf("3.prime\n");
        printf("4.Exit\n");
        printf("Enter your choice ");
        scanf("%d",&ch);
switch(ch)
{
        case 1: pthread_create(&tid1,NULL,fib,NULL);
                pthread_join(tid1,NULL);
                pthread_cancel(tid1);
                break;
        case 2: pthread_create(&tid2,NULL,fact,NULL);
                pthread_join(tid2,NULL);
                pthread_cancel(tid2);
                break;
        case 3: pthread_create(&tid3,NULL,prime,NULL);
                pthread_join(tid3,NULL);
                pthread_cancel(tid2);
                break;
        case 4: printf("Exit Successfully!\n");
                return EXIT_SUCCESS;
        default:
               printf("Invalid choice");
               break;
}
        ch++;
        }

return 0;
}
