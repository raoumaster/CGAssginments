#!/bin/bash
# Menu driven program for find command
echo -e "1.check for the file emp.txt\n2.files modified before 3 days\n3.searching for files ending with .txt/.dat contains string capgemini\n4.searching regular files in home\n5.searching for emp.dat in root, error messages redirect to null\n6.exit"
read -p "Enter your choice: " choice
case $choice in
        1) find -name emp.txt;;
        2) find -iname "*.sh" -mtime -3 -print;;
        3)
                #find . -type f \( -name "*.sh" -o -name "*.txt" \) | grep -rnw -e 'capgemini';;
                grep --include=\*.{txt,sh} -rnw -e "capgemini";;
        4) find . /home type f;;
        5) find  /root -name emp.dat &> /dev/null;;
        6) exit;;



esac
