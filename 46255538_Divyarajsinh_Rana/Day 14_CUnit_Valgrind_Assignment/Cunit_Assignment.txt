#include <stdio.h>
#include <stdlib.h>
#include "CUnit/CUnit.h"
#include "CUnit/Basic.h"    

#define MIN(x, y)
    x < y ? x :y

#define MAX(x, y)
    x > y ? x :y

//find maximum in given array
int getMax(int arr[], int n)
{
	int res = arr[0];
	int i = 0;
	for (i = 1; i < n; i++)
		res = MAX(res, arr[i]);
	return res;
}

char* MaxLen(char* s1, char* s2)
{
	if(strlen(s1) > strlen(s2))
		return 1;
	else
		return s2;
}

//find minimum in given array
int getMin(int arr[], int n)
{
	int res = arr[0];
	int i = 0;
	for (i = 1; i < n; i++)
		res = MIN(res, arr[i]);
	return res;
}

// test cases to compare the strings

void test_MaxLen_0(void)
{
    CU_ASSERT_EQUAL(MaxLen("Code", "Program"), "Program");
}

void test_MaxLen_1(void)
{
    CU_ASSERT_EQUAL(MaxLen("Hockey", "Cricket"), "Cricket");
}

void test_MaxLen_2(void)
{
    CU_ASSERT_EQUAL(MaxLen("Java", "Python"), "Python");
}

void test_MaxLen_3(void)
{
    CU_ASSERT_EQUAL(MaxLen("Bank", "Customer"), "Customer");
}

void test_MaxLen_4(void)
{
    CU_ASSERT_EQUAL(MaxLen("Bow", "Arrow"), "Bow");
}

void test_MaxLen_5(void)    
{
    CU_ASSERT_EQUAL(MaxLen("Autopilot", "Tesla"), "Autopilot");
}

void test_MaxLen_6(void)
{
    CU_ASSERT_EQUAL(MaxLen("India", "Gandhinagar"), "India");
}

int main()
{
	int arr[]={0,0,0,0,0}; // Initializing the array elements with 0 value
  int min = getMin(arr, sizeof(arr)/sizeof(arr[0])); // Function call for minimum
  int max = getMax(arr, sizeof(arr)/sizeof(arr[0]));  // Function call for maximum
  printf("\n min: %d max: %d\n", min, max);

	CU_initialize_registry();
	CU_pSuite MaxLenSuite = CU_add_suite("MaxLen", NULL, NULL);   
	CU_add_test(MaxLenSuite, "MaxLen('Code', 'Program')",test_MaxLen_0);
	CU_add_test(MaxLenSuite, "MaxLen('Hockey', 'Cricket')",test_MaxLen_1);
	CU_add_test(MaxLenSuite, "MaxLen('Java', 'Python')",test_MaxLen_2);
	CU_add_test(MaxLenSuite, "MaxLen('Bank', 'Customer')",test_MaxLen_3);
	CU_add_test(MaxLenSuite, "MaxLen('Bow', 'Arrow')",test_MaxLen_4);
	CU_add_test(MaxLenSuite, "MaxLen('Autopilot', 'Tesla')",test_MaxLen_5);
	CU_add_test(MaxLenSuite, "MaxLen('India', 'Gandhinagar')",test_MaxLen_6);
	CU_basic_run_tests();  
	CU_cleanup_registry();
	return 0;
}
