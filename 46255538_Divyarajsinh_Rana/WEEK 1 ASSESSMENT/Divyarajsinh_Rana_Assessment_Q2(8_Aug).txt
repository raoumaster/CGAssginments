//Including header files

#include<stdio.h>
#define R 4
#define C 4

//function to print matrix in spiral clockwise
void spiral_clockwise(int m, int n, int a[R][C])
{
        int i,k=0,l=0;
        while(k<m && l<n)
        {
                //first row
                for(i=l; i<n; ++i)
                {
                        printf("%d", a[k][i]);
                }
                k++;

                //last column
                for(i=k; i<m; ++i)
                {
                        printf("%d", a[i][n-1]);
                }
                n--;

                //last row
                if(k<m)
                {
                        for(i=n-1; i>=l; --i)
                        {
                                printf("%d",a[m-1][i]);
                        }
                        m--;
                }

                //first column
                if(l<n)
                {
                        for(i=m-1; i>=k; --i){
                                printf("%d",a[i][l]);
                        }
                        l++;
                }
        }
}
