Q2) Prompt the user to enter an upper case letter and display the letter in the lowercase.


#include <stdio.h>
int main ()
{
        char upperCase;
        int lowerCase;

        printf("Enter the character : ");
        scanf("%c", &upperCase);
        lowerCase = upperCase + 32;
        printf("Character in lower case is: %c\n", lowerCase);
        return 0;
}
