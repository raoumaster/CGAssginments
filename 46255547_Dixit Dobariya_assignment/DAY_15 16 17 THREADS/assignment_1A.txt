
#include<stdio.h>
#include<stdlib.h>
#include<pthread.h>
#include<unistd.h>
#include<string.h>
#define MAX 100

/*TSYN /02 TSYN /02	Thread1 should count words and return to caller.	
Hint:Modify startroutine1() to receive a line of text and count words and return to caller.
*/

void* startRoutine1(void* str){
        int count=1;
        char *stp1 = (char *) str;
        for(int i=0;stp1[i] != '\0'; i++){
                if(stp1[i] == ' ' && stp1[i+1] != ' '){
                        count++;
                }
        }
        printf("the no of words in the line is %d\n",count);
}
// TSYN /03 Thread 2 to count characters and return to caller

void* startRoutine2(void* str){
        int count = 0;
        char *stp1 = (char *) str;
        for(int i=0; stp1[i] != '\0'; i++){
                count++;
        }
        printf("the no of characters in the line is %d\n",count);
}

int main(){
        char str[MAX];
        fgets(str,MAX,stdin);
        pthread_t thread1,thread2;
        pthread_create(&thread1,NULL,&startRoutine1,str);
        pthread_create(&thread2,NULL,&startRoutine2,str);
        pthread_join(thread1,NULL);
        pthread_join(thread2,NULL);
}
